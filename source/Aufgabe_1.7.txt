# include < iostream >

//int var ist vor einer Methode definiert worden und somit im ganzen Dokument erreichbar.
int var = 3;

//Double a und b 
double sum ( double a , double b )
{
    return a + b ;
}
int square ( int var )
{
    return var * var ;
}

{

        std :: cout << " i ^2 = " << square ( i ) << ’\ n ’;
        std :: cout << " i + i = " << sum (i , i ) << ’\ n ’;
    }
    return 0;
}
